// 核心算法：利用 charAt(） 遍历这个字符串 //把每个字符都存储给对象， 如果对象没有该属性，就为1，如果存在了就 +1 //遍历对象，得到最大值和该字符 var str = 'daskljahiagnaknscsjdwfanalvnahdpwod'; var o = {}; for (var i = 0; i < str.length; i++) { var chars = str.charAt(i); // chars 是 字符串的每一个字符 if (o[chars]) { // o[chars] 得到的是属性值 o[chars]++; } else { o[chars] = 1; } } console.log(o); // 2. 遍历对象 var max = 0; var ch = ''; for (var k in o) { // k 得到是 属性名 // o[k] 得到的是属性值 if (o[k] > max) { max = o[k]; ch = k; } } alert(max); alert('最多的字符是' + ch);
